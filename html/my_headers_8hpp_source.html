95// Cheb nodes
96Eigen::VectorXd cheb_nodes(double a, double b, int n)
97{
98 Eigen::VectorXd X(n);
99 double l, l1, param;
100 l = 0.5 * (a + b);
101 l1 = 0.5 * (b - a);
102 for (int k = 0; k < n; k++)
103 {
104 param = (double)(k + 0.5) / n;
105 X(k) = l - l1 * cos(param * 3.1412);
106 }
107 return X;
108}

110// Uniform nodes
111Eigen::VectorXd uniform_nodes(double a, double b, int n)
112{
113 Eigen::VectorXd X(n);
114 double l = (b-a);
115 for (int k = 0; k < n; k++)
116 {
117 X(k) = a + l/(double) (n-1) * k;
118 }
119 return X;
120}
